Example JavaScript function and regular expression

function isS4TDL(str) {
	let rv={};
	const regex = RegExp( "^https?:(?:\/\/(?:(?:(?:[^\/?#]*)@)?(?:[^\/?#:]*)(?::(?:[^\/?#]*))?))?(?:(?:[^?#]*)\/(?:00|sscc)\/)([0-9]{18})" );
	let matches=regex.exec(str);
	rv.isS4T = regex.test(str);
	if (rv.isS4T) {
		rv.sscc = matches[1];
	} else {
		rv.sscc=null;
	}
	return rv;
}
